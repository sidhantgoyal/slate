Add BuyNow Button Code

You can add the BuyNow button to your view like you would any other UIView. Create a new BuyNow button WMTBuyNowControl instance along with the productIds as shown in the code example shown below.

  • Objective_C:

// ViewController.m example
#import "ViewController.h"
@import WMTSDKKit;

@interface ViewController()
@end

@implementation ViewController

- (void)viewDidLoad {
    [super viewDidLoad];
    NSSet *productIdSet = [NSSet setWithObjects:@"39665237", @"47364217", nil];
    WMTBuyNowControl *buyNowButton = [[WMTBuyNowControl alloc] initWithProductIds:productIdSet];
    buyNowButton.center = self.view.center;
    [self.view addSubview:buyNowButton];
}

@end
  • Swift:

// ViewController.swift example
import UIKit
import WMTSDKKit

class ViewController: UIViewController {

    override func viewDidLoad() {
        super.viewDidLoad()
        let productIdsSet: Set<String> = ["39665237", "47364217"]
        let wmtBuyNowControl = WMTBuyNowControl(productIds: productIdsSet)
        wmtBuyNowControl.center = self.view.center
        self.view.addSubview(wmtBuyNowControl)
    }
}
